      // Render cards
      function renderCards() {
        giftCards.forEach((card, i) => {
          const btn = document.createElement('button');
          btn.classList.add('card');
          btn.setAttribute('aria-label', `Select ${card.name} gift card`);
          btn.style.setProperty('--bg-url', `url(${card.bg})`);
          // set background via ::before using style attribute:
          btn.style.position = 'relative';

          btn.style.backgroundImage = `url(${card.bg})`;
          btn.style.backgroundSize = 'cover';
          btn.style.backgroundPosition = 'center center';

          btn.innerHTML = `

              <h3>${card.name}</h3>
              <p>Select to choose your gift card</p>
            </div>
          `;

          btn.addEventListener('click', () => openPopup(card));
          cardsContainer.appendChild(btn);
        });
      }

      // Open amounts popup
      function openPopup(card) {
        selectedCard = card;
        selectedAmount = null;
        claimBtn.disabled = true;
        generatedCodeEl.style.display = 'none';
        generatedCodeEl.textContent = '';

        // Clear previous amount buttons
        amountButtonsContainer.innerHTML = '';

        // Create amount buttons
        amounts.forEach(amount => {
          const btn = document.createElement('button');
          btn.classList.add('amount-btn');
          btn.textContent = `$${amount}`;
          btn.addEventListener('click', () => {
            selectedAmount = amount;
            claimBtn.disabled = false;
            // Highlight selected amount button
            [...amountButtonsContainer.children].forEach(b => b.classList.remove('selected'));
            btn.classList.add('selected');
            generatedCodeEl.style.display = 'none';
            generatedCodeEl.textContent = '';
          });
          amountButtonsContainer.appendChild(btn);
        });

        popup.querySelector('#popup-title').textContent = `Select Amount for ${card.name}`;
        popup.classList.add('active');
        popup.focus();
      }

      // Generate fake gift code
      function generateCode() {
        const chars = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789';
        let code = '';
        for (let i = 0; i < 16; i++) {
          if (i > 0 && i % 4 === 0) code += '-';
          code += chars.charAt(Math.floor(Math.random() * chars.length));
        }
        return code;
      }

      // Claim button click
      claimBtn.addEventListener('click', () => {
        if (!selectedCard || !selectedAmount) return;
        // Show generated code with blur effect first
        generatedCodeEl.textContent = generateCode();
        generatedCodeEl.style.display = 'block';
        generatedCodeEl.classList.remove('revealed');

        setTimeout(() => {
          generatedCodeEl.classList.add('revealed');
        }, 600);

        // After 3 seconds redirect to offer link
        setTimeout(() => {
          window.location.href = selectedCard.offerLink;
        }, 3000);
      });

      // Close popup
      closeBtn.addEventListener('click', () => {
        popup.classList.remove('active');
      });

      // Close on outside click
      popup.addEventListener('click', (e) => {
        if (e.target === popup) {
          popup.classList.remove('active');
        }
      });

      // Close on ESC key
      document.addEventListener('keydown', (e) => {
        if (e.key === 'Escape' && popup.classList.contains('active')) {
          popup.classList.remove('active');
        }
      });

      // Initialize page
      renderCards();
